import modal
import io, json, re, time

app = modal.App("chequescan-ai")

image = (
    modal.Image.debian_slim(python_version="3.11")
    .apt_install(["tesseract-ocr", "tesseract-ocr-ben"])
    .pip_install(["pillow", "paddleocr", "google-generativeai", "gradio"])
)

BANGLADESH_BANKS = {
    "dutch_bangla": {"code": "090", "patterns": ["dbbl", "ডাচ-বাংলা"]},
    "brac": {"code": "060", "patterns": ["brac", "ব্র্যাক"]},
    # ... 60 more banks
}

@app.cls(image=image, secrets=[modal.Secret.from_name("google-secret")])
class ChequeScan:
    
    @modal.enter()
    def setup(self):
        import google.generativeai as genai
        from paddleocr import PaddleOCR
        
        genai.configure(api_key=os.environ["GOOGLE_API_KEY"])
        self.gemini = genai.GenerativeModel('gemini-2.0-flash-exp')
        self.ocr = PaddleOCR(lang='en', use_gpu=False)
    
    @modal.method()
    def process(self, image_bytes: bytes) -> dict:
        from PIL import Image
        
        image = Image.open(io.BytesIO(image_bytes))
        
        # OCR extraction
        ocr_result = self.ocr.ocr(np.array(image))
        text = " ".join([l[1][0] for l in ocr_result[0]])
        
        # VLM analysis
        prompt = f"Extract from Bangladesh cheque: {text[:500]}"
        response = self.gemini.generate_content([prompt, image])
        
        return json.loads(re.search(r'\{.*\}', response.text).group())
