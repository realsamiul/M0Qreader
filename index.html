chequescan.py 
import modal
import io, json, re, time, base64, hashlib
from typing import Dict

app = modal.App("chequescan-ai-prod")

image = (
    modal.Image.debian_slim(python_version="3.11")
    .apt_install(["tesseract-ocr", "tesseract-ocr-ben", "libgl1-mesa-glx"])
    .pip_install([
        "pillow", "opencv-python-headless", "numpy",
        "paddleocr", "paddlepaddle", "google-generativeai",
        "gradio", "fastapi"
    ])
)

# Bangladesh Bank Database
BANGLADESH_BANKS = {
    "dutch_bangla": {"code": "090", "patterns": ["dutch-bangla", "dbbl", "ডাচ-বাংলা"]},
    "brac": {"code": "060", "patterns": ["brac", "ব্র্যাক"]},
    "city": {"code": "225", "patterns": ["city bank", "সিটি"]},
    # ... 59 more banks
}

@app.cls(image=image, secrets=[modal.Secret.from_name("google-secret")], memory=4096)
class ChequeScanProcessor:
    
    @modal.enter()
    def initialize(self):
        import google.generativeai as genai
        from paddleocr import PaddleOCR
        import os
        
        genai.configure(api_key=os.environ["GOOGLE_API_KEY"])
        self.gemini = genai.GenerativeModel('gemini-2.0-flash-exp')
        self.ocr = PaddleOCR(use_angle_cls=True, lang='en', use_gpu=False)
        self.cache = {}
    
    def preprocess(self, image_bytes: bytes):
        import cv2
        import numpy as np
        from PIL import Image
        
        img = Image.open(io.BytesIO(image_bytes))
        img_np = np.array(img.convert('RGB'))
        gray = cv2.cvtColor(img_np, cv2.COLOR_RGB2GRAY)
        
        # Quality assessment
        blur_score = cv2.Laplacian(gray, cv2.CV_64F).var()
        quality = "A" if blur_score > 300 else "B" if blur_score > 100 else "C"
        
        # Adaptive enhancement
        if blur_score < 200:
            img = cv2.bilateralFilter(img_np, 9, 75, 75)
        
        return img, {"quality": quality, "blur": blur_score}
    
    def analyze_with_gemini(self, image, ocr_text: str) -> Dict:
        prompt = f"""Analyze Bangladesh bank cheque. OCR: {ocr_text[:500]}
Extract: bank_name, account_number, cheque_number, date, payee_name, 
amount_figures, amount_words. Return JSON with confidence scores."""
        
        response = self.gemini.generate_content([prompt, image])
        match = re.search(r'\{.*\}', response.text, re.DOTALL)
        return json.loads(match.group()) if match else {}
    
    @modal.method()
    def process(self, image_bytes: bytes) -> Dict:
        start = time.time()
        
        # Check cache
        img_hash = hashlib.md5(image_bytes).hexdigest()[:16]
        if img_hash in self.cache:
            return {**self.cache[img_hash], "cache_hit": True}
        
        # Process
        image, quality = self.preprocess(image_bytes)
        ocr_result = self.ocr.ocr(np.array(image))
        ocr_text = " ".join([line[1][0] for line in ocr_result[0]] if ocr_result[0] else [])
        extraction = self.analyze_with_gemini(image, ocr_text)
        
        result = {
            "status": "success",
            "extraction": extraction,
            "quality": quality,
            "time_ms": int((time.time() - start) * 1000)
        }
        
        self.cache[img_hash] = result
        return result

@app.function(image=image, secrets=[modal.Secret.from_name("google-secret")])
@modal.asgi_app()
def web():
    import gradio as gr
    from fastapi import FastAPI
    
    def process_upload(image):
        buf = io.BytesIO()
        image.save(buf, format='PNG')
        processor = ChequeScanProcessor()
        return json.dumps(processor.process.remote(buf.getvalue()), indent=2)
    
    demo = gr.Interface(
        fn=process_upload,
        inputs=gr.Image(type="pil"),
        outputs=gr.Code(language="json"),
        title="🏦 ChequeScan AI"
    )
    
    return gr.mount_gradio_app(FastAPI(), demo, path="/")

# Call API
