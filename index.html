import modal
import io, json, re, time

app = modal.App("chequescan-ai")

image = (
    modal.Image.debian_slim(python_version="3.11")
    .apt_install(["tesseract-ocr", "tesseract-ocr-ben"])
    .pip_install(["pillow", "paddleocr", "google-generativeai", "gradio"])
)

BANGLADESH_BANKS = {
    "IFIC Bank": {"code": "IFIC", "swift": "IFICBDDH"},
    "City Bank": {"code": "CITY", "swift": "CITYBDDH"},
    # ... more banks
}

class ChequeProcessor(modal.cls(image=image)):
    def __enter__(self):
        from paddleocr import PaddleOCR
        import google.generativeai as genai
        self.ocr = PaddleOCR(use_angle_cls=True, lang='en', ocr_version='PP-OCRv4')
        genai.configure(api_key=modal.secret.from_name("google-secret"))
        self.gemini = genai.GenerativeModel('gemini-pro-vision')

    @modal.method()
    def process(self, image_bytes: bytes):
        image = Image.open(io.BytesIO(image_bytes))
        ocr_result = self.ocr.ocr(image, cls=True)
        text = " ".join([l[1][0] for l in ocr_result[0]])

        # VLM analysis
        prompt = f"Extract from Bangladesh cheque: {text[:500]}"
        response = self.gemini.generate_content([prompt, image])
        return json.loads(re.search(r'\{.*\}', response.text).group())
